JavaScript学习笔记(手写promise)

2021-07-29 JavaScript

废话不多说, 直接上代码.

class MyPromise {

  static PENDING = "pending"

  static FULFILLED = "fulfilled"

  static REJECTED = "rejected"

  constructor(executor) {
    this.status = MyPromise.PENDING
    this.value = null
    this.callbacks = []
    try { // 捕获执行者出现错误
      executor(this.resovle.bind(this), this.reject.bind(this)) // 绑定this
    } catch (error) {
      this.reject(error)
    }
  }

  resovle(value) {
    if (this.status === MyPromise.PENDING) { // 状态保护
      this.status = MyPromise.FULFILLED
      this.value = value
      setTimeout(() => {
        this.callbacks.map(callback => {
          callback.onFulfilled(value)
        })
      })
    }
  }

  reject(reason) {
    if (this.status == MyPromise.PENDING) { // 状态保护
      this.status = MyPromise.REJECTED
      this.value = reason
      setTimeout(() => {
        this.callbacks.map(callback => {
          callback.onRejected(reason)
        })
      })
    }
  }

  then(onFulfilled, onRejected) {
    if (typeof onFulfilled === 'function') {
      onFulfilled = () => this.value
    }

    if (typeof onRejected === 'function') {
      onRejected = () => this.value
    }

    let promise = new MyPromise((resovle, reject) => {
      if (this.status === MyPromise.PENDING) {
        this.callbacks.push({
          onFulfilled: value => {
            this.parse(promise, onFulfilled(value), resovle, reject)
          },
          onRejected: reason => {
            this.parse(promise, onRejected(reason), resovle, reject)
          }
        })
      }

      if (this.status === MyPromise.FULFILLED) {
        setTimeout(() => {
          this.parse(promise, onFulfilled(this.value), resovle, reject)
        })
      }

      if (this.status === MyPromise.REJECTED) {
        setTimeout(() => {
          this.parse(promise, onRejected(this.value), resovle, reject)
        })
      }
    })

    return promise
  }

  parse(promise, result, resovle, reject) {
    if (promise == result) {
      throw TypeError("Chaining cycle detected for MyPromise")
    }
    try {
      if (result instanceof MyPromise) {
        result.then(resovle, reject)
        // result.then(value => {
        //   resovle(result)
        // }, reason => {
        //   reject(reason)
        // })
      } else {
        resovle(result)
      }
    } catch (error) {
      reject(error)
    }
  }

  static resovle(value) {
    return new MyPromise((resovle, reject) => {
      if (value instanceof MyPromise) {
        value.then(resovle, reject)
      } else {
        resovle(value)
      }
    })
  }

  static reject(reason) {
    return new MyPromise((resovle, reject) => {
      reject(reason)
    })
  }

  static all(promises) {
    const values = []
    return new MyPromise((resolve, reject) => {
      promises.forEach(promise => {
        promise.then(value => {
          values.push(value)
          if (values.length === promies.length) {
            resolve()
          }
        }, reason => {
          reject(reason)
        })
      })
    })
  }

  static race(promises) {
    return new MyPromise((resovle, reject) => {
      promises.map(promise => {
        promise.then(value => {
          resovle(value)
        }, reason => {
          reject(reason)
        })
      })
    })
  }
}
